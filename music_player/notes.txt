    isPlaying = !isPlaying
    const playBtn = document.querySelector('.play-btn')
    const bgVideo = document.getElementById('bg-video')

    if (isPlaying) {
        playBtn.textContent = '⏸'
        audio.play()

    } else {

    }
}
